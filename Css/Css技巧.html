  height: 400px;
  width: 600px;
  background-color: antiquewhite;
  display: flex;
  justify-content: center;
  align-items: center;
}
.container:empty::after {
  display: block;
  position: absolute;
  content: "暂无数据";
  top: 50%;
  left: 50%;
  transform: translate(-50% -50%);
}
