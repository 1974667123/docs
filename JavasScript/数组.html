# 数组常用方法

# 1.Map （）方法返回一个新数组，数组中的元素为原始数组元素调用函数处理后的值。

let arr1 = [0, 1, 2, 3, 1, 5];
let a = arr1.map((item) => {
  return item + 1;
});
console.log(a); // [1, 2, 3, 4, 2, 6]

​ map() 方法按照原始数组元素顺序依次处理元素。注意： map() 不会改变原始数组 注意： map() 不会改变原始数组。

​

# 2.For 循环对象

# for in 循环遍历的是数组的键值(索引)，而 for of 循环遍历的是数组的值

（1）For...in 循环 循环对象（ x 循环的是对象的属性）

var person = { fname: "John", lname: "Doe", age: 25 };

var text = "";
var x;
for (x in person) {
  text += person[x] + " ";
}
//结果：John Doe 25

（2）For...of 循环 （循环 x 是对象的值）

let person = { name: "老王", age: 23, city: "唐山" };
for (let item of person) {
  console.log(item);
} //结果：老王 23 唐山

# 2.For 循环数组

# （1）for 循环可以使用break，return，continue 跳出循环，但**forEach 不能，只能通过 try …catch 来终止

continue 跳过本次循环
return   跳出当前循环（跳出距离最近的循环）
break    跳出全部的循环

# （2）forEach 循环中，如果需要跳出整个循环，需要 throw 抛异常

try {
  arr.forEach((a, i) => {
    if (i === 2) {
      throw new Error();
    }
    console.log("forEach===throw", a);
  });
} catch (e) {
  console.log(e);
}

# 3.unshift() 方法可向数组的开头添加一个或更多元素

var fruits = ["Banana", "Orange", "Apple", "Mango"];
fruits.unshift("Lemon", "Pineapple");
//Lemon,Pineapple,Banana,Orange,Apple,Mango

# 4.push() 方法可向数组的末尾添加一个或多个元素

var fruits = ["Banana", "Orange", "Apple", "Mango"];
fruits.push("Kiwi");
//Banana,Orange,Apple,Mango,Kiwi

# 5.some() 方法用于检测数组中的元素 是否有满足指定条件

var ages = [3, 10, 18, 20];
//只要有一个符合就可以，就返回true  其余的不检测 不改变原数组，形成新数组
ages.some((el) => {
  return el >= 18;
}); // true
  • 如果有一个元素满足条件，则表达式返回true , 剩余的元素不会再执行检测。
  • 如果没有满足条件的元素，则返回 false。

# 6.every() 方法用于检测数组所有元素是否都符合指定条件

var ages = [32, 33, 16, 40];
//检测数组 是否所有元素都符合条件
ages.every((el) => {
  return age >= 18;
}); //false

# 7.filter() 方法创建一个新的数组，筛选符合条件的元素

var ages = [32, 33, 16, 40];

function checkAdult(age) {
  return age >= 18;
}

let arr = ages.filter(checkAdult); //[32, 33, 40]

# 8.reduce() 方法接收一个函数作为累加器

var numbers = [65, 44, 12, 4];
numbers.reduce((total, num) => {
  return total + num;
});
// total:初始值 0 num：每一项的值

# 9.findIndex() 方法 接受一个方法 返回符合条件第一次出现的下标。

var numbers = [65, 44, 12, 4];
let index = numbers.findIndex((el) => {
  return el > 50;
}); // index :0
//当数组中的元素在测试条件时返回 true 时, findIndex() 返回符合条件的元素的索引位置，之后的值不会再调用执行函数。
//如果没有符合条件的元素返回 -1

# 10.includes() 方法用于判断字符串是否包含指定的子字符串

//如果找到匹配的字符串则返回 true，否则返回 false。
var str = "Hello world, welcome to the Runoob。";
var n = str.includes("Runoob"); //true

//也可以是数组

# 11.sort() 方法用于对数组的元素进行排序

let arr = [0, 1, 5, 35, 12, 3, 13, 11];
function compare(val1, val2) {
  if (val1 < val2) {
    return -1;
  } else if (val1 > val2) {
    return 1;
  } else {
    return 0;
  }
}
arr.sort(compare); //[0, 1, 3, 5, 11, 12, 13, 35]

# 11.js-数组嵌套另一个数组

var arr1 = ["n1", "n2", "n3"];
var arr2 = ["n4", "n5", "n6"];
var arr = [];
arr.push(arr1);
arr.push(arr2);
//[["n1","n2","n3"],["n4","n5","n6"]]
