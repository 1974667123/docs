      this.$nextTick(() => {
        this.dragging = null;
        this.ending = null;
      });
    },
    // 拖动到元素上触发
    handleDragOver(e) {
      // 首先把div变成可以放置的元素，即重写dragenter/dragover
      // 在dragenter中针对放置目标来设置
      e.dataTransfer.dropEffect = "move";
    },
    // 拖进时
    handleDragEnter(e, item) {
      // 为需要移动的元素设置dragstart事件
      e.dataTransfer.effectAllowed = "move";
      // 记录目标元素
      this.ending = item;
    },
  },
};
