# 2.通过 JS 获取可视窗口的高度

this.$nextTick(() => {
  this.tableHeight = window.innerHeright - 235;
});

window.onresize = () => {
  this.$nextTick(() => {
    this.tableHeight = window.innerHeright - 235;
  });
};
